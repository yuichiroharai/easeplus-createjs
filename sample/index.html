Sample - Ease+ for TweenJS(CreateJS)

Expo

  • expoOut

    createjs.Tween.get(shape).to({x:450}, 800, createjs.Ease.expoOut);

  • expoIn

    createjs.Tween.get(shape).to({x:450}, 800, createjs.Ease.expoIn);

  • expoInOut

    createjs.Tween.get(shape).to({x:450}, 800, createjs.Ease.expoInOut);

Blend

// イージング関数をブレンドして新たなイージング関数を取得
// 引数は
// ・イージング1
// ・イージング2
// ・イージング1から2に移行する比率を調整するイージング(デフォルトはlinear)
var Ease = createjs.Ease;
createjs.Tween.get(shape)
  .to({x:450}, 600, Ease.blend(Ease.cubicIn, Ease.backOut, Ease.sineOut));
